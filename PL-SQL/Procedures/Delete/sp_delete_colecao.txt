
CREATE OR REPLACE PROCEDURE sp_delete_colecao(

vCliente cliente.cd_cliente%type

 )   IS 

-- Nomes
vApelido cliente.nome_ex_cliente%type;

 BEGIN
     
    SELECT nome_ex_cliente into vApelido from cliente where cd_cliente = vCliente; 
    
     DELETE FROM Colecao
     WHERE cd_cliente = vCliente;
        
     dbms_output.put_line('Coleção de [' || vApelido  || '] apagada com sucesso!');
     COMMIT;
         
    EXCEPTION            
       
        WHEN no_data_found THEN 
        
             RAISE_APPLICATION_ERROR(-20001,'Erro ao tentar apagar coleção, verifique se o código  [' || vCliente|| '] é válido.');   
          
        WHEN OTHERS THEN
            
             RAISE_APPLICATION_ERROR(-20003, 'Erro desconhecido ao tentar apagar coleção de [' || vApelido|| ']');
                  
 END sp_delete_colecao;
